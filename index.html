-- STEP 4: Create Clean Table--

 

DROP TABLE IF EXISTS used_cars_clean;

 

CREATE TABLE used_cars_clean AS

 

SELECT

 

    CAST(latitude AS DOUBLE)                    AS latitude,

 

    CAST(longitude AS DOUBLE)                   AS longitude,

 

    city,

 

    YEAR(TO_DATE(listed_date))                  AS listing_year,

 

    MONTH(TO_DATE(listed_date))                 AS listing_month,

 

    CAST(daysonmarket AS INT)                   AS days_on_market,

 

    CASE

 

        WHEN CAST(daysonmarket AS INT) <= 20 THEN 'High Demand'

 

        WHEN CAST(daysonmarket AS INT) <= 45 THEN 'Medium Demand'

 

        ELSE 'Low Demand'

 

    END                                         AS demand_category,

 

    make_name,

 

    fuel_type,

 

    CAST(price AS DOUBLE)                       AS price,

 

    CAST(mileage AS DOUBLE)                     AS mileage,

 

    CASE

 

        WHEN has_accidents = 'True' THEN true

 

        ELSE false

 

    END                                         AS has_accidents

 

FROM used_cars_raw

 

WHERE

 

    price IS NOT NULL

 

    AND CAST(price AS DOUBLE) > 500

 

    AND latitude IS NOT NULL

 

    AND longitude IS NOT NULL

 

    AND listed_date IS NOT NULL

 

    AND make_name IS NOT NULL

 

    AND daysonmarket IS NOT NULL

 

    AND fuel_type IS NOT NULL

 

    AND has_accidents IS NOT NULL

 

    AND mileage IS NOT NULL

 

    AND CAST(mileage AS DOUBLE) > 0

 

LIMIT 500000;

 

 

 

-- Verify clean table--

 

SELECT * FROM used_cars_clean LIMIT 5;

DESCRIBE FORMATTED used_cars_clean;

 

 

 

 

-- QUERY 1  : Demand by Fuel Type Over Time--

 

SELECT

 

    TRIM(fuel_type)                 AS fuel_type,

 

    listing_year,

 

    COUNT(*)                        AS total_listings,

 

    ROUND(AVG(days_on_market), 1)   AS avg_days_on_market

 

FROM used_cars_clean

 

WHERE

 

    listing_year >= 2016

 

    AND fuel_type IS NOT NULL

 

    AND TRIM(fuel_type) <> ''

 

    AND days_on_market IS NOT NULL

 

GROUP BY

 

    TRIM(fuel_type),

 

    listing_year

 

HAVING

 

    COUNT(*) >= 50

 

ORDER BY

 

    listing_year,

 

    total_listings DESC

 

LIMIT 20;

 

 

-- QUERY 2  : Accident History Impact on Price and Demand--

 

SELECT

 

    has_accidents,

 

    COUNT(*)                        AS total_listings,

 

    ROUND(AVG(price), 2)            AS avg_price,

 

    ROUND(AVG(days_on_market), 1)   AS avg_days_on_market

 

FROM used_cars_clean

 

GROUP BY

 

    has_accidents

 

ORDER BY

 

    avg_price DESC;

 

 

 

-- QUERY 3  : Top 10 Used Car Brands in the US--

-- EXPORT QUERY 6:  Average price by fuel type and mileage

 

INSERT OVERWRITE DIRECTORY '/tmp/teremio/fuel_mileage'

 

ROW FORMAT DELIMITED FIELDS TERMINATED BY ','

 

SELECT

 

    TRIM(fuel_type)                 AS fuel_type,

 

    CASE

 

        WHEN mileage < 20000  THEN '0-20k'

 

        WHEN mileage < 40000  THEN '20k-40k'

 

        WHEN mileage < 60000  THEN '40k-60k'

 

        WHEN mileage < 80000  THEN '60k-80k'

 

        WHEN mileage < 100000 THEN '80k-100k'

 

        ELSE '100k+'

 

    END                             AS mileage_bucket,

 

    COUNT(*)                        AS total_listings,

 

    ROUND(AVG(price), 2)            AS avg_price,

 

    ROUND(AVG(mileage), 0)          AS avg_mileage

 

FROM used_cars_clean

 

WHERE

 

    mileage IS NOT NULL

 

    AND mileage > 0

 

    AND fuel_type IS NOT NULL

 

    AND TRIM(fuel_type) <> ''

 

    AND TRIM(fuel_type) <> 'N/A'

 

GROUP BY

 

    TRIM(fuel_type),

 

    CASE

 

        WHEN mileage < 20000  THEN '0-20k'

 

        WHEN mileage < 40000  THEN '20k-40k'

 

        WHEN mileage < 60000  THEN '40k-60k'

 

        WHEN mileage < 80000  THEN '60k-80k'

 

        WHEN mileage < 100000 THEN '80k-100k'

 

        ELSE '100k+'

 

    END

 

ORDER BY

 

    fuel_type,

 

    avg_mileage ASC;
